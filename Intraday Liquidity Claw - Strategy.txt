var int vp_zoneStart = 0
int vp_lookback = bar_index - vp_zoneStart
var bool vp_activeZone = false
var vp_vpGreen = array.new_float(vpResolution, 0)
var vp_vpRed = array.new_float(vpResolution, 0)
var vp_zoneBounds = array.new_float(vpResolution, 0)
var float[] vp_ltfOpen = array.new_float(0)
var float[] vp_ltfClose = array.new_float(0)
var float[] vp_ltfHigh = array.new_float(0)
var float[] vp_ltfLow = array.new_float(0)
var float[] vp_ltfVolume = array.new_float(0)

string userSymbol = syminfo.prefix + ":" + syminfo.ticker
string openInterestTicker = str.format("{0}_OI", userSymbol)
string timeframe_oi = syminfo.type == "futures" and timeframe.isintraday ? "1D" : timeframe.period
deltaOi = request.security(openInterestTicker, timeframe_oi, close - close[1], ignore_invalid_symbol=true)

vp_vol() =>
    out = vpSmoothVol ? ta.ema(volume, 5) : volume
    if vpVolType == 'Open Interest'
        out := deltaOi
    out

[vp_dO, vp_dC, vp_dH, vp_dL, vp_dV] = request.security_lower_tf(syminfo.tickerid, '', [open, close, high, low, vp_vol()])

vp_resetProfile(bool enable) =>
    if enable
        array.fill(vp_vpGreen, 0)
        array.fill(vp_vpRed, 0)
        array.clear(vp_ltfOpen)
        array.clear(vp_ltfHigh)
        array.clear(vp_ltfLow)
        array.clear(vp_ltfClose)
        array.clear(vp_ltfVolume)

vp_profHigh = ta.highest(high, vp_lookback + 1)[1]
vp_profLow  = ta.lowest(low, vp_lookback + 1)[1]

vp_get_vol(float y11, float y12, float y21, float y22, float height, float vol) =>
    nz(math.max(math.min(math.max(y11, y12), math.max(y21, y22)) - math.max(math.min(y11, y12), math.min(y21, y22)), 0) * vol / height)

vp_profileAdd(float o, float h, float l, float c, float v, float g) =>
    for i = 0 to array.size(vp_vpGreen) - 1
        zoneTop = array.get(vp_zoneBounds, i)
        zoneBot = zoneTop - g
        body_top = math.max(c, o)
        body_bot = math.min(c, o)
        itsgreen = c >= o
        topwick = h - body_top
        bottomwick = body_bot - l
        body = body_top - body_bot
        bodyvol = body * v / (2 * topwick + 2 * bottomwick + body)
        topwickvol = 2 * topwick * v / (2 * topwick + 2 * bottomwick + body)
        bottomwickvol = 2 * bottomwick * v / (2 * topwick + 2 * bottomwick + body)
        if vpVolType == 'Volume'
            array.set(vp_vpGreen, i, array.get(vp_vpGreen, i) + (itsgreen ? vp_get_vol(zoneBot, zoneTop, body_bot, body_top, body, bodyvol) : 0) + vp_get_vol(zoneBot, zoneTop, body_top, h, topwick, topwickvol) / 2 + vp_get_vol(zoneBot, zoneTop, body_bot, l, bottomwick, bottomwickvol) / 2)
            array.set(vp_vpRed, i, array.get(vp_vpRed, i) + (itsgreen ? 0 : vp_get_vol(zoneBot, zoneTop, body_bot, body_top, body, bodyvol)) + vp_get_vol(zoneBot, zoneTop, body_top, h, topwick, topwickvol) / 2 + vp_get_vol(zoneBot, zoneTop, body_bot, l, bottomwick, bottomwickvol) / 2)
        else if vpVolType == 'Open Interest'
            if v > 0
                array.set(vp_vpGreen, i, array.get(vp_vpGreen, i) + vp_get_vol(zoneBot, zoneTop, body_bot, body_top, body, v))
            if v < 0
                array.set(vp_vpRed, i, array.get(vp_vpRed, i) + vp_get_vol(zoneBot, zoneTop, body_bot, body_top, body, -v))

vp_calcSession(bool update) =>
    array.fill(vp_vpGreen, 0)
    array.fill(vp_vpRed, 0)
    if bar_index > vp_lookback and update
        gap = (vp_profHigh - vp_profLow) / vpResolution
        for i = 0 to vpResolution - 1
            array.set(vp_zoneBounds, i, vp_profHigh - gap * i)

vp_pocLevel() =>
    float maxVol = 0
    int levelInd = 0
    for i = 0 to array.size(vp_vpRed) - 1
        if array.get(vp_vpRed, i) + array.get(vp_vpGreen, i) > maxVol
            maxVol := array.get(vp_vpRed, i) + array.get(vp_vpGreen, i)
            levelInd := i
    float outLevel = na
    if levelInd != array.size(vp_vpRed) - 1
        outLevel := array.get(vp_zoneBounds, levelInd) - (array.get(vp_zoneBounds, levelInd) - array.get(vp_zoneBounds, levelInd + 1)) / 2
    outLevel

vp_valueLevels(float poc) =>
    float gap = (vp_profHigh - vp_profLow) / vpResolution
    float volSum = array.sum(vp_vpRed) + array.sum(vp_vpGreen)
    float volCnt = 0
    float vah = vp_profHigh
    float val = vp_profLow
    int pocInd = 0
    for i = 0 to array.size(vp_zoneBounds) - 2
        if array.get(vp_zoneBounds, i) >= poc and array.get(vp_zoneBounds, i + 1) < poc
            pocInd := i
    volCnt += array.get(vp_vpRed, pocInd) + array.get(vp_vpGreen, pocInd)
    for i = 1 to array.size(vp_vpRed)
        if pocInd + i >= 0 and pocInd + i < array.size(vp_vpRed)
            volCnt += array.get(vp_vpRed, pocInd + i) + array.get(vp_vpGreen, pocInd + i)
            if volCnt >= volSum * (vpVAwidth / 100)
                break
            else
                val := array.get(vp_zoneBounds, pocInd + i) - gap
        if pocInd - i >= 0 and pocInd - i < array.size(vp_vpRed)
            volCnt += array.get(vp_vpRed, pocInd - i) + array.get(vp_vpGreen, pocInd - i)
            if volCnt >= volSum * (vpVAwidth / 100)
                break
            else
                vah := array.get(vp_zoneBounds, pocInd - i)
    result = array.new<float>(2)
    array.set(result, 0, val)
    array.set(result, 1, vah)
    result

calcLiqTrail(mlen, alen, amult) =>
    float t_atr = ta.atr(alen)
    float t_ma  = ta.ema(close, mlen)
    float raw_up = t_ma - t_atr * amult
    float raw_dn = t_ma + t_atr * amult
    int t_trend = 1
    float t_trail = na
    if barstate.isfirst
        t_trend := close > t_ma ? 1 : -1
        t_trail := close > t_ma ? raw_up : raw_dn
    else
        int prev_trend = nz(t_trend[1], 1)
        float prev_trail = nz(t_trail[1], close)
        if prev_trend == 1
            t_trail := math.max(raw_up, prev_trail)
            t_trend := close < t_trail ? -1 : 1
            if close < t_trail
                t_trail := raw_dn
        else
            t_trail := math.min(raw_dn, prev_trail)
            t_trend := close > t_trail ? 1 : -1
            if close > t_trail
                t_trail := raw_up
    [t_trend, t_trail]
